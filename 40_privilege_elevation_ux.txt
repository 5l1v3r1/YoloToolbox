# Operating System, kernel version, or service pack info
````
cat /etc/issue
cat /etc/*release
cat /proc/version
ls /boot | grep "vmlinuz"
lsb_release -a
uname -a
````
 
Find user information
````
id
whoami
last
````
 
Check installed programs, permissions, and hidden files
````
ls -lah
ls -lah /usr/bin
ls -lah /sbin
yum list installed
dpkg-query -l
rpm -qa
ls -lah /usr/share/applications | awk -F '.desktop' ' { print $1}'
````
 
Manual escalation commands
````
sudo su
sudo -i
sudo /bin/bash
sudo su-
sudo ht
pkexec visudo
/etc/passwd
/etc/sudoers
find / \( -perm -2000 -o -perm -4000 \) -exec ls -ld {} \; 2>/dev/null
find / -type d \( -perm -g+w -or -perm -o+w \) -exec ls -adl {} \;
````

Evaluate running services
````
ps aux
ps aux -u root
systemctl status (service)
top
pstree
cat /etc/services
service --status-all
````

 
Check for scheduled tasks/jobs
````
cat /etc/cron.d/*
cat /var/spool/cron/*
crontab -l
cat /etc/crontab
cat /etc/cron.(time)
systemctl list-timers
````
